class Solution {
    public int search(int[] nums, int target) {
        int result=-1;
        int right=0;
            int mid;
            System.out.println(nums.length);
        int left=nums.length-1;
        while(right<=left){
            mid=left+right/2;
            if(target==nums[mid]){
            
                
                return mid;
            }
            
            else if(target>nums[mid]){
                right=mid+1;
            }
            else{
                left=mid-1;
            }
            
        }
        if(right>left){
            return result;
        }
        return result;
        
        
        
    }
}
